  <files>src/api/useAuth.ts</files>
  <action>
    ```ts
    import { useEffect, useState } from 'react'
    import type { User } from '@supabase/supabase-js'
    import { supabase } from '../lib/supabase'

    export type AuthState = {
      user: User | null
      isLoading: boolean
    }

    let cachedState: AuthState | null = null

    export function useAuth() {
      const [state, setState] = useState<AuthState>(
        () => cachedState ?? { user: null, isLoading: true },
      )

      useEffect(() => {
        let mounted = true
        // 1. Lê sessão inicial
        supabase.auth.getSession().then(({ data }) => {
          if (!mounted) return
          const next = { user: data.session?.user ?? null, isLoading: false }
          cachedState = next
          setState(next)
        })

        // 2. Subscribe pra mudanças (login, logout, refresh)
        const { data: sub } = supabase.auth.onAuthStateChange((_event, session) => {
          if (!mounted) return
          const next = { user: session?.user ?? null, isLoading: false }
          cachedState = next
          setState(next)
        })

        return () => {
          mounted = false
          sub.subscription.unsubscribe()
        }
      }, [])

      return state
    }

    export async function signInWithEmail(email: string): Promise<{ error?: string }> {
      const { error } = await supabase.auth.signInWithOtp({
        email: email.trim().toLowerCase(),
        options: {
          emailRedirectTo: window.location.origin,
        },
      })
      if (error) return { error: error.message }
      return {}
    }

    export async function signOut(): Promise<void> {
      await supabase.auth.signOut()
    }
    ```

    Avoid:
    - Usar React Query aqui (não cabe — sessão é estado client-side, não dado de server)
    - Passar redirectTo dinâmico baseado em pathname (`emailRedirectTo: window.location.origin` resolve magic links pra raiz e o guard cuida do resto)
