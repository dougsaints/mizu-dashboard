```sql
do $$
declare
  tbl text;
  tables text[] := array[
    'sales_daily', 'ads_daily', 'ads_imports', 'anotaai_imports',
    'anotaai_products', 'daily_entries', 'dashboard_widgets',
    'data_sources', 'diary_entries', 'organic_entries',

  foreach tbl in array tables loop
    execute format(
      'create policy phase1_anon_all on public.%I as permissive for all using (true) with check (true)',
      tbl

  end loop;
end $$;

create policy phase1_anon_select_tenants on public.tenants

  using (true);
```
